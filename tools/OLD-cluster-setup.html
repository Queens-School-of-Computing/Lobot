kind: ConfigMap
metadata:
  name: time-slicing-config-fine
data:
  a16: |-
    version: v1
    flags:
      migStrategy: none
    sharing:
      timeSlicing:
        resources:
        - name: nvidia.com/gpu
          replicas: 4
